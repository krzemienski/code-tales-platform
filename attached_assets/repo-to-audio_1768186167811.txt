@dataclass
class RepositoryContent:
    """Extracted repository content and metadata"""
    url: str
    name: str
    full_content: str
    file_tree: str
    total_files: int
    total_lines: int
    primary_languages: List[str]
    readme: str
    key_files: Dict[str, str]  # Important files content
    dependencies: Dict[str, Any]
    extracted_at: datetime
    content_hash: str
    chunks: List[str] = field(default_factory=list)  # For large repos


class RepomixExtractor:
    """Handles repository content extraction using repomix CLI"""
    
    def __init__(self, cache_dir: str = ".repo_audio_cache"):
        self.cache_dir = Path(cache_dir)
        self.cache_dir.mkdir(exist_ok=True)
        self._verify_repomix_installed()
    
    def _verify_repomix_installed(self):
        """Verify repomix is installed"""
        try:
            result = subprocess.run(
                ["repomix", "--version"],
                capture_output=True,
                text=True
            )
            if result.returncode != 0:
                raise RuntimeError("repomix not found")
            logger.info(f"repomix version: {result.stdout.strip()}")
        except FileNotFoundError:
            raise RuntimeError(
                "repomix is not installed. Install with: npm install -g repomix"
            )
    
    def _get_cache_path(self, url: str) -> Path:
        """Get cache file path for a repository"""
        url_hash = hashlib.sha256(url.encode()).hexdigest()[:16]
        return self.cache_dir / f"repo_{url_hash}.json"
    
    def _load_from_cache(self, url: str) -> Optional[RepositoryContent]:
        """Try to load repository content from cache"""
        cache_path = self._get_cache_path(url)
        if cache_path.exists():
            try:
                with open(cache_path, 'r') as f:
                    data = json.load(f)
                    data['extracted_at'] = datetime.fromisoformat(data['extracted_at'])
                    return RepositoryContent(**data)
            except Exception as e:
                logger.warning(f"Cache load failed: {e}")
        return None
    
    def _save_to_cache(self, content: RepositoryContent):
        """Save repository content to cache"""
        cache_path = self._get_cache_path(content.url)
        data = asdict(content)
        data['extracted_at'] = content.extracted_at.isoformat()
        with open(cache_path, 'w') as f:
            json.dump(data, f)
    
    async def extract(
        self,
        url: str,
        use_cache: bool = True,
        chunk_size: int = 100000
    ) -> RepositoryContent:
        """Extract repository content using repomix"""
        
        # Check cache first
        if use_cache:
            cached = self._load_from_cache(url)
            if cached:
                logger.info(f"Loaded from cache: {url}")
                return cached
        
        console.print(Panel(f"[bold blue]Extracting repository: {url}[/bold blue]"))
        
        with tempfile.TemporaryDirectory() as temp_dir:
            output_file = Path(temp_dir) / "repomix-output.txt"
            
            # Run repomix
            with Progress(
                SpinnerColumn(),
                TextColumn("[progress.description]{task.description}"),
                console=console
            ) as progress:
                task = progress.add_task("Running repomix...", total=None)
                
                # Build repomix command
                cmd = [
                    "repomix",
                    "--remote", url,
                    "--output", str(output_file),
                    "--style", "plain",
                    "--include-priority-files",
                ]
                
                process = await asyncio.create_subprocess_exec(
                    *cmd,
                    stdout=asyncio.subprocess.PIPE,
                    stderr=asyncio.subprocess.PIPE,
                    cwd=temp_dir
                )
                
                stdout, stderr = await process.communicate()
                
                if process.returncode != 0:
                    error_msg = stderr.decode() if stderr else "Unknown error"
                    raise RuntimeError(f"repomix failed: {error_msg}")
                
                progress.update(task, completed=True)
            
            # Read output
            with open(output_file, 'r', encoding='utf-8', errors='ignore') as f:
                full_content = f.read()
        
        # Parse repository name from URL
        repo_name = url.rstrip('/').split('/')[-1].replace('.git', '')
        
        # Extract key sections from repomix output
        file_tree = self._extract_section(full_content, "File Tree", "Files")
        readme = self._extract_readme(full_content)
        key_files = self._extract_key_files(full_content)
        dependencies = self._extract_dependencies(full_content)
        languages = self._detect_languages(full_content)
        
        # Count files and lines
        total_files = full_content.count("## File: ")
        total_lines = len(full_content.split('\n'))
        
        # Create content hash
        content_hash = hashlib.sha256(full_content.encode()).hexdigest()[:16]
        
        # Chunk content for large repos
        chunks = []
        if len(full_content) > chunk_size:
            chunks = self._chunk_content(full_content, chunk_size)
        
        content = RepositoryContent(
            url=url,
            name=repo_name,
            full_content=full_content,
            file_tree=file_tree,
            total_files=total_files,
            total_lines=total_lines,
            primary_languages=languages,
            readme=readme,
            key_files=key_files,
            dependencies=dependencies,
            extracted_at=datetime.now(),
            content_hash=content_hash,
            chunks=chunks
        )
        
        # Cache the result
        if use_cache:
            self._save_to_cache(content)
        
        return content
    
    def _extract_section(self, content: str, start_marker: str, end_marker: str) -> str:
        """Extract a section from repomix output"""
        try:
            start = content.find(start_marker)
            end = content.find(end_marker, start + len(start_marker))
            if start != -1 and end != -1:
                return content[start:end].strip()
        except Exception:
            pass
        return ""
    
    def _extract_readme(self, content: str) -> str:
        """Extract README content"""
        patterns = ["## File: README.md", "## File: readme.md", "## File: README"]
        for pattern in patterns:
            idx = content.find(pattern)
            if idx != -1:
                # Find next file or end
                next_file = content.find("## File:", idx + len(pattern))
                if next_file != -1:
                    return content[idx:next_file].strip()
                return content[idx:idx + 10000].strip()
        return ""
    
    def _extract_key_files(self, content: str) -> Dict[str, str]:
        """Extract important configuration and entry files"""
        key_patterns = [
            "package.json", "setup.py", "pyproject.toml", "Cargo.toml",
            "go.mod", "pom.xml", "build.gradle", "Makefile", "Dockerfile",
            "docker-compose.yml", ".env.example", "config.yaml", "config.json"
        ]
        key_files = {}
        for pattern in key_patterns:
            marker = f"## File: {pattern}"
            idx = content.find(marker)
            if idx == -1:
                # Try with path variations
                for alt in [f"## File: ./{pattern}", f"## File: src/{pattern}"]:
                    idx = content.find(alt)
                    if idx != -1:
                        break
            
            if idx != -1:
                next_file = content.find("## File:", idx + len(marker))
                if next_file != -1:
                    key_files[pattern] = content[idx:next_file].strip()[:5000]
                else:
                    key_files[pattern] = content[idx:idx + 5000].strip()
        return key_files
    
    def _extract_dependencies(self, content: str) -> Dict[str, Any]:
        """Extract and parse dependency information"""
        deps = {"detected": [], "raw": {}}
        
        # Check for package.json
        if "package.json" in content:
            try:
                # Find package.json content
                idx = content.find('"dependencies"')
                if idx != -1:
                    deps["detected"].append("npm")
            except Exception:
                pass
        
        # Check for Python
        if any(x in content for x in ["requirements.txt", "setup.py", "pyproject.toml"]):
            deps["detected"].append("python")
        
        # Check for other package managers
        if "Cargo.toml" in content:
            deps["detected"].append("rust/cargo")
        if "go.mod" in content:
            deps["detected"].append("go")
        if "pom.xml" in content or "build.gradle" in content:
            deps["detected"].append("java")
        
        return deps
    
    def _detect_languages(self, content: str) -> List[str]:
        """Detect primary programming languages"""
        language_patterns = {
            ".py": "Python",
            ".js": "JavaScript",
            ".ts": "TypeScript",
            ".jsx": "React/JavaScript",
            ".tsx": "React/TypeScript",
            ".rs": "Rust",
            ".go": "Go",
            ".java": "Java",
            ".cpp": "C++",
            ".c": "C",
            ".rb": "Ruby",
            ".php": "PHP",
            ".swift": "Swift",
            ".kt": "Kotlin",
        }
        
        detected = []
        for ext, lang in language_patterns.items():
            if f"## File:" in content and ext in content:
                count = content.count(f"{ext}\n") + content.count(f"{ext} ")
                if count > 0:
                    detected.append((lang, count))
        
        # Sort by count and return top languages
        detected.sort(key=lambda x: x[1], reverse=True)
        return [lang for lang, _ in detected[:5]]
    
    def _chunk_content(self, content: str, chunk_size: int) -> List[str]:
        """Split content into manageable chunks for processing"""
        chunks = []
        
        # Try to split on file boundaries
        files = content.split("## File: ")
        current_chunk = ""
        
        for file_content in files:
            if len(current_chunk) + len(file_content) > chunk_size:
                if current_chunk:
                    chunks.append(current_chunk)
                current_chunk = file_content
            else:
                current_chunk += "\n## File: " + file_content if current_chunk else file_content
        
        if current_chunk:
            chunks.append(current_chunk)
        
        return chunks

    async def execute(
        self,
        task: str,
        context: Optional[str] = None,
        previous_results: Optional[List[AgentResult]] = None
    ) -> AgentResult:
        """Execute the agent's task"""
        
        import time
        start_time = time.time()
        
        # Build the full prompt
        messages = []
        
        # Add context from previous agents if available
        if previous_results:
            context_parts = ["## Previous Agent Outputs:\n"]
            for result in previous_results:
                if result.success:
                    context_parts.append(f"### {result.agent_role.value}:\n{result.output[:5000]}\n")
            if context:
                context = "\n".join(context_parts) + "\n\n" + context
            else:
                context = "\n".join(context_parts)
        
        # Build user message
        user_message = task
        if context:
            user_message = f"## Context:\n{context}\n\n## Task:\n{task}"
        
        messages.append({"role": "user", "content": user_message})
        
        try:
            # Use extended thinking for complex analysis
            response = self.client.messages.create(
                model=self.model,
                max_tokens=self.max_tokens,
                temperature=1,  # Required for extended thinking
                thinking={
                    "type": "enabled",
                    "budget_tokens": 10000  # Allow deep reasoning
                },
                system=self.system_prompt,
                messages=messages
            )
            
            # Extract output and reasoning
            output = ""
            reasoning = ""
            
            for block in response.content:
                if block.type == "thinking":
                    reasoning = block.thinking
                elif block.type == "text":
                    output = block.text
            
            duration = time.time() - start_time
            
            return AgentResult(
                agent_role=self.role,
                success=True,
                output=output,
                reasoning=reasoning,
                tokens_used=response.usage.input_tokens + response.usage.output_tokens,
                duration_seconds=duration
            )
            
        except Exception as e:
            logger.error(f"Agent {self.role.value} failed: {e}")
            return AgentResult(
                agent_role=self.role,
                success=False,
                output=str(e),
                duration_seconds=time.time() - start_time
            )


class AgentOrchestrator:
    """Coordinates multiple Claude agents for content production"""
    
    def __init__(
        self,
        config: PipelineConfig,
        anthropic_client: anthropic.Anthropic
    ):
        self.config = config
        self.client = anthropic_client
        self.agents: Dict[AgentRole, ClaudeAgent] = {}
        self.results: List[AgentResult] = []
        self._initialize_agents()
    
    def _initialize_agents(self):
        """Initialize agents based on output format and configuration"""
        
        # Determine which agents to use based on format
        required_agents = self._get_required_agents()
        
        for role in required_agents:
            custom_instructions = self.config.custom_instructions.custom_prompts.get(role)
            
            # Add global context to all agents
            if self.config.custom_instructions.global_context:
                if custom_instructions:
                    custom_instructions = f"{self.config.custom_instructions.global_context}\n\n{custom_instructions}"
                else:
                    custom_instructions = self.config.custom_instructions.global_context
            
            self.agents[role] = ClaudeAgent(
                client=self.client,
                role=role,
                model=self.config.claude_model,
                temperature=self.config.agent_temperature,
                custom_instructions=custom_instructions
            )
    
    def _get_required_agents(self) -> List[AgentRole]:
        """Determine which agents are needed based on output format"""
        
        # Base agents for all formats
        base_agents = [

        format_specific = {

                AgentRole.SCRIPT_REVIEWER,

                AgentRole.SCRIPT_REVIEWER,

    async def run_analysis_phase(
        self,
        repo_content: RepositoryContent
    ) -> List[AgentResult]:
        """Run all analysis agents in parallel where possible"""
        
        console.print("\n[bold cyan]🔍 Phase 1: Repository Analysis[/bold cyan]")
        
        # Analysis agents can run in parallel
        analysis_agents = [
            AgentRole.ARCHITECTURE_ANALYST,
            AgentRole.CODE_DEEP_DIVE,
            AgentRole.DOCUMENTATION_ANALYST,
            AgentRole.DEPENDENCY_ANALYST,
        ]
        
        # Filter to only agents we have
        parallel_agents = [a for a in analysis_agents if a in self.agents]
        
        # Build context for analysis
        context = self._build_analysis_context(repo_content)
        
        # Add custom analysis instructions
        if self.config.custom_instructions.analysis_instructions:
            context += f"\n\n## Custom Analysis Instructions:\n{self.config.custom_instructions.analysis_instructions}"
        
        tasks = []
        for role in parallel_agents:
            task_prompt = self._get_analysis_task(role, repo_content)
            tasks.append(self.agents[role].execute(task_prompt, context))
        
        # Run in parallel with progress display
        with Progress(
            SpinnerColumn(),
            TextColumn("[progress.description]{task.description}"),
            BarColumn(),
            TextColumn("{task.completed}/{task.total}"),
            console=console
        ) as progress:
            task = progress.add_task("Running analysis agents...", total=len(tasks))
            
            results = []
            for coro in asyncio.as_completed(tasks):
                result = await coro
                results.append(result)
                self.results.append(result)
                progress.advance(task)
                
                status = "✅" if result.success else "❌"
                console.print(f"  {status} {result.agent_role.value}: {result.tokens_used} tokens")
        
        return results
    
    async def run_creative_phase(
        self,
        analysis_results: List[AgentResult]
    ) -> List[AgentResult]:
        """Run creative/narrative agents"""
        
        console.print("\n[bold magenta]🎨 Phase 2: Creative Development[/bold magenta]")
        
        creative_agents = [
            AgentRole.NARRATIVE_ARCHITECT,
            AgentRole.CHARACTER_DESIGNER,
            AgentRole.WORLD_BUILDER,
            AgentRole.DIALOGUE_WRITER,
        ]
        
        # Filter to only agents we have
        active_agents = [a for a in creative_agents if a in self.agents]
        
        results = []
        
        for role in active_agents:
            task_prompt = self._get_creative_task(role)
            
            with console.status(f"[bold yellow]Running {role.value}...[/bold yellow]"):
                result = await self.agents[role].execute(
                    task_prompt,
                    previous_results=analysis_results + results
                )
            
            results.append(result)
            self.results.append(result)
            
            status = "✅" if result.success else "❌"
            console.print(f"  {status} {role.value}: {result.tokens_used} tokens")
        
        return results
    
    async def run_production_phase(
        self,
        previous_results: List[AgentResult]
    ) -> AgentResult:
        """Run script writing and review"""
        
        console.print("\n[bold green]📝 Phase 3: Script Production[/bold green]")
        
        # Script writer
        if AgentRole.SCRIPT_WRITER in self.agents:
            task_prompt = self._get_script_task()
            
            # Add custom script instructions
            if self.config.custom_instructions.script_instructions:
                task_prompt += f"\n\n## Custom Instructions:\n{self.config.custom_instructions.script_instructions}"
            
            with console.status("[bold yellow]Writing script...[/bold yellow]"):
                script_result = await self.agents[AgentRole.SCRIPT_WRITER].execute(
                    task_prompt,
                    previous_results=previous_results
                )
            
            self.results.append(script_result)
            console.print(f"  ✅ Script written: {script_result.tokens_used} tokens")
            
            # Script reviewer
            if AgentRole.SCRIPT_REVIEWER in self.agents:
                review_task = """Review and improve the script for:
1. Technical accuracy
2. Engagement and pacing
3. Natural dialogue flow
4. Appropriate length
5. Clear speaker attribution

Provide the revised script with improvements incorporated."""
                
                with console.status("[bold yellow]Reviewing script...[/bold yellow]"):
                    review_result = await self.agents[AgentRole.SCRIPT_REVIEWER].execute(
                        review_task,
                        previous_results=[script_result]
                    )
                
                self.results.append(review_result)
                console.print(f"  ✅ Script reviewed: {review_result.tokens_used} tokens")
                
                return review_result
            
            return script_result
        
        raise RuntimeError("No script writer agent configured")
    
    async def run_synthesis_phase(

    ) -> AgentResult:
        """Final synthesis of all outputs"""

class ElevenLabsGenFMGenerator:
    """Uses ElevenLabs GenFM to generate scripts and audio directly from content"""
    
    def __init__(self, client: ElevenLabs, config: PipelineConfig):
        self.client = client
        self.config = config
    
    async def generate_from_url(self, url: str) -> Dict[str, Any]:
        """Generate podcast directly from URL using GenFM"""
        
        console.print(Panel("[bold cyan]Using ElevenLabs GenFM for generation[/bold cyan]"))
        
        # Map our config to ElevenLabs parameters
        duration_map = {
            ContentLength.MICRO: "short",
            ContentLength.SHORT: "short",
            ContentLength.MEDIUM: "default",
            ContentLength.LONG: "long",
            ContentLength.EXTENDED: "long",
            ContentLength.FULL: "long",
        }
        
        mode_type = "conversational" if self.config.output_format == OutputFormat.PODCAST else "bulletin"
        
        # Note: This requires Studio API access from ElevenLabs
        try:
            podcast = self.client.studio.create_podcast(
                safety_identifier=f"repo-podcast-{datetime.now().timestamp()}",
                model_id=self.config.production_settings.model_id,
                mode={
                    "type": mode_type,
                    "num_speakers": self.config.podcast_config.num_speakers if self.config.podcast_config else 2
                },
                source={
                    "type": "url",
                    "url": url
                },
                duration=duration_map.get(ContentLength(self.config.length), "default"),
                quality=self.config.production_settings.quality,
                callback_url=self.config.callback_url
            )
            
            return {
                "project_id": podcast.project_id if hasattr(podcast, 'project_id') else None,
                "status": "created",
                "method": "genfm"
            }
            
        except Exception as e:
            logger.error(f"GenFM generation failed: {e}")
            raise
    
    async def generate_from_content(self, content: str, title: str) -> Dict[str, Any]:
        """Generate from text content using ElevenLabs Studio"""
        
        console.print(Panel("[bold cyan]Creating ElevenLabs Studio project[/bold cyan]"))
        
        try:
            # Create project from content
            project = self.client.studio.create_project(
                name=title,
                from_content=content,
                default_model_id=self.config.production_settings.model_id,
                quality=self.config.production_settings.quality,
                auto_assign_voices=self.config.auto_assign_voices,
                callback_url=self.config.callback_url
            )
            
            return {
                "project_id": project.project_id if hasattr(project, 'project_id') else None,
                "status": "created",
                "method": "studio_project"
            }
            
        except Exception as e:
            logger.error(f"Studio project creation failed: {e}")
            raise

class AudioSynthesizer:
    """Handles audio synthesis using ElevenLabs Studio API"""
    
    def __init__(self, client: ElevenLabs, config: PipelineConfig):
        self.client = client
        self.config = config
    
    def _parse_script_to_content(self, script: str) -> List[Dict[str, Any]]:
        """Parse script into ElevenLabs content structure"""
        
        chapters = []
        current_chapter = {"name": "Introduction", "blocks": []}
        
        lines = script.split('\n')
        
        for line in lines:
            line = line.strip()
            if not line:
                continue
            
            # Chapter markers
            if line.startswith('###'):
                if current_chapter["blocks"]:
                    chapters.append(current_chapter)
                current_chapter = {
                    "name": line.replace('#', '').strip(),
                    "blocks": []
                }
                continue
            
            # Speaker lines (SPEAKER: text)
            speaker_match = re.match(r'^([A-Z_]+):\s*(.+)$', line)
            if speaker_match:
                speaker = speaker_match.group(1)
                text = speaker_match.group(2)
                
                # Remove stage directions from spoken text
                text = re.sub(r'\[.*?\]', '', text).strip()
                
                if text:
                    # Find voice for speaker
                    voice_id = self._get_voice_for_speaker(speaker)
                    
                    current_chapter["blocks"].append({
                        "sub_type": "p",
                        "nodes": [{
                            "type": "tts_node",
                            "voice_id": voice_id,
                            "text": text
                        }]
                    })
            else:
                # Narration or description
                if not line.startswith('['):  # Skip pure stage directions
                    text = re.sub(r'\[.*?\]', '', line).strip()
                    if text:
                        voice_id = self._get_narrator_voice()
                        current_chapter["blocks"].append({
                            "sub_type": "p",
                            "nodes": [{
                                "type": "tts_node",
                                "voice_id": voice_id,
                                "text": text
                            }]
                        })
        
        # Add final chapter
        if current_chapter["blocks"]:
            chapters.append(current_chapter)
        
        return chapters
    
    def _get_voice_for_speaker(self, speaker: str) -> str:
        """Get voice ID for a speaker"""
        
        # Check configured voices
        for voice in self.config.voices:
            if voice.name.upper() == speaker.upper() or voice.role.upper() == speaker.upper():
                return voice.voice_id
        
        # Default voices (these are example IDs - replace with real ones)
        default_voices = {
            "HOST": "21m00Tcm4TlvDq8ikWAM",  # Rachel
            "EXPERT": "AZnzlk1XvdvUeBnXmlld",  # Domi
            "NARRATOR": "EXAVITQu4vr4xnSDxMaL",  # Bella
            "GUEST": "MF3mGyEYCl7XYWbV9V6O",  # Elli
        }
        
        return default_voices.get(speaker, default_voices["NARRATOR"])
    
    def _get_narrator_voice(self) -> str:
        """Get the default narrator voice"""
        for voice in self.config.voices:
            if voice.role.lower() == "narrator":
                return voice.voice_id
        return "EXAVITQu4vr4xnSDxMaL"  # Default Bella voice
    
    async def synthesize(
        self,
        script: str,
        title: str,
        output_path: Optional[str] = None
    ) -> Dict[str, Any]:
        """Synthesize script to audio using ElevenLabs Studio API"""
        
        console.print(Panel("[bold green]🎙️ Synthesizing Audio[/bold green]"))
        
        # Parse script to content structure
        content = self._parse_script_to_content(script)
        
        if not content:
            raise ValueError("No content parsed from script")
        
        console.print(f"  📚 Parsed {len(content)} chapters")
        
        try:
            # Create Studio project
            with console.status("[bold yellow]Creating Studio project...[/bold yellow]"):
                project = self.client.studio.add_project(
                    name=title,
                    from_content=json.dumps(content),
                    default_model_id=self.config.production_settings.model_id,
                    quality=self.config.production_settings.quality.value
                    if hasattr(self.config.production_settings.quality, 'value')
                    else self.config.production_settings.quality,
                    callback_url=self.config.callback_url
                )
            
            project_id = project.project_id
            console.print(f"  ✅ Project created: {project_id}")
            
            # Get chapters and convert
            with console.status("[bold yellow]Converting chapters...[/bold yellow]"):
                chapters = self.client.studio.get_chapters(project_id)
                
                for chapter in chapters.chapters:
                    self.client.studio.convert_chapter(
                        project_id=project_id,
                        chapter_id=chapter.chapter_id
                    )
            
            console.print(f"  ✅ Conversion started for {len(chapters.chapters)} chapters")
            
            # If output path specified, we'd poll and download
            # For now, return the project info
            result = {
                "project_id": project_id,
                "chapters": len(content),
                "status": "converting",
                "method": "studio_api"
            }
            
            if output_path:
                result["output_path"] = output_path
                console.print(f"  📁 Audio will be saved to: {output_path}")
                console.print("  ⏳ Use the project_id to poll for completion and download")
            
            return result
            
        except Exception as e:
            logger.error(f"Audio synthesis failed: {e}")
            raise

class RepoToAudioPipeline:
    """Main pipeline orchestrating the entire process"""
    
    def __init__(self, config: PipelineConfig):
        self.config = config
        
        # Initialize clients
        self.anthropic_client = anthropic.Anthropic(
            api_key=os.environ.get("ANTHROPIC_API_KEY")
        )
        self.elevenlabs_client = ElevenLabs(
            api_key=os.environ.get("ELEVENLABS_API_KEY")
        )
        
        # Initialize components
        self.extractor = RepomixExtractor(cache_dir=config.cache_dir)
        self.orchestrator = AgentOrchestrator(config, self.anthropic_client)
        self.genfm_generator = ElevenLabsGenFMGenerator(self.elevenlabs_client, config)
        self.synthesizer = AudioSynthesizer(self.elevenlabs_client, config)
    
    async def run(self) -> Dict[str, Any]:
        """Execute the full pipeline"""
        
        console.print(Panel.fit(
            f"[bold]Repository to Audio Pipeline[/bold]\n"
            f"Source: {self.config.source_url}\n"
            f"Format: {self.config.output_format}\n"
            f"Length: {self.config.length}\n"
            f"Path: {self.config.script_generation_path}",
            title="🎙️ Starting Pipeline"
        ))
        
        result = {
            "config": asdict(self.config) if hasattr(self.config, '__dataclass_fields__') else self.config.dict(),
            "stages": {},
            "success": False
        }
        
        try:
            # Path 1: ElevenLabs GenFM (direct URL to podcast)
            if self.config.script_generation_path == ScriptGenerationPath.ELEVENLABS_GENFM:
                return await self._run_genfm_path()
            
            # Path 2 & 3: Claude Agents (with or without ElevenLabs enhancement)
            return await self._run_claude_path()
            
        except Exception as e:
            logger.error(f"Pipeline failed: {e}")
            result["error"] = str(e)
            raise
    
    async def _run_genfm_path(self) -> Dict[str, Any]:
        """Run the ElevenLabs GenFM direct path"""
        
        console.print("\n[bold cyan]Using ElevenLabs GenFM Direct Path[/bold cyan]")
        
        # For GitHub repos, we might want to use the repo's README or docs URL
        # Or use repomix output as content
        
        if "github.com" in self.config.source_url:
            # Extract repo content first
            repo_content = await self.extractor.extract(
                self.config.source_url,
                use_cache=self.config.enable_caching
            )
            
            # Use GenFM with the content
            result = await self.genfm_generator.generate_from_content(
                content=repo_content.readme + "\n\n" + repo_content.full_content[:50000],
                title=f"Exploring {repo_content.name}"
            )
        else:
            # Direct URL
            result = await self.genfm_generator.generate_from_url(self.config.source_url)
        
        console.print(Panel("[bold green]✅ GenFM Generation Complete[/bold green]"))
        return result
    
    async def _run_claude_path(self) -> Dict[str, Any]:
        """Run the Claude agents path"""
        
        result = {"stages": {}}
        
        # Stage 1: Extract repository content
        console.print("\n[bold]📦 Stage 1: Repository Extraction[/bold]")
        repo_content = await self.extractor.extract(
            self.config.source_url,
            use_cache=self.config.enable_caching,
            chunk_size=self.config.chunk_size
        )
        
        result["stages"]["extraction"] = {
            "name": repo_content.name,
            "files": repo_content.total_files,
            "lines": repo_content.total_lines,
            "languages": repo_content.primary_languages,
            "chunks": len(repo_content.chunks)
        }
        
        console.print(f"  ✅ Extracted: {repo_content.total_files} files, {repo_content.total_lines} lines")
        
        # Stage 2: Analysis Phase
        analysis_results = await self.orchestrator.run_analysis_phase(repo_content)
        result["stages"]["analysis"] = {
            "agents": len(analysis_results),
            "successful": sum(1 for r in analysis_results if r.success),
            "total_tokens": sum(r.tokens_used for r in analysis_results)
        }
        
        # Stage 3: Creative Phase
        creative_results = await self.orchestrator.run_creative_phase(analysis_results)
        result["stages"]["creative"] = {
            "agents": len(creative_results),
            "successful": sum(1 for r in creative_results if r.success),
            "total_tokens": sum(r.tokens_used for r in creative_results)
        }
        
        # Stage 4: Production Phase
        script_result = await self.orchestrator.run_production_phase(
            analysis_results + creative_results
        )
        result["stages"]["production"] = {
            "success": script_result.success,
            "tokens": script_result.tokens_used
        }
        
        # Stage 5: Final Synthesis
        final_result = await self.orchestrator.run_synthesis_phase(
            self.orchestrator.results
        )
        result["stages"]["synthesis"] = {
            "success": final_result.success,
            "tokens": final_result.tokens_used
        }
        
        # Store the script
        result["script"] = final_result.output
        
        # Stage 6: Audio Synthesis (if not hybrid mode, or if hybrid)
        if self.config.script_generation_path in [ScriptGenerationPath.CLAUDE_AGENTS, ScriptGenerationPath.HYBRID]:
            audio_result = await self.synthesizer.synthesize(
                script=final_result.output,
                title=f"Exploring {repo_content.name}",
                output_path=self.config.output_path
            )
            result["stages"]["audio"] = audio_result
        
        result["success"] = True
        
        # Print summary
        self._print_summary(result)
        
        return result
    
    def _print_summary(self, result: Dict[str, Any]):
        """Print a summary of the pipeline execution"""
        
        table = Table(title="Pipeline Summary")
        table.add_column("Stage", style="cyan")
        table.add_column("Status", style="green")
        table.add_column("Details", style="yellow")
        
        for stage_name, stage_data in result.get("stages", {}).items():
            if isinstance(stage_data, dict):
                status = "✅" if stage_data.get("success", True) else "❌"
                details = ", ".join(f"{k}: {v}" for k, v in stage_data.items() if k != "success")
            else:
                status = "✅"
                details = str(stage_data)
            
            table.add_row(stage_name.title(), status, details[:50])
        
        console.print("\n")
        console.print(table)

def load_config_from_yaml(path: str) -> Dict[str, Any]:
    """Load configuration from YAML file"""
    with open(path, 'r') as f:
        return yaml.safe_load(f)


def create_config_from_args(args) -> PipelineConfig:
    """Create PipelineConfig from CLI arguments"""
    
    config_dict = {
        "source_url": args.url,
        "output_format": args.format,
        "length": args.length,
        "tone": args.tone,
        "audience_level": args.audience,
        "script_generation_path": args.path,
        "output_path": args.output,
    }
    
    # Load from YAML if provided
    if args.config:
        yaml_config = load_config_from_yaml(args.config)
        config_dict.update(yaml_config)
    
    # Add custom instructions if provided
    if args.instructions:
        config_dict["custom_instructions"] = CustomInstructions(
            global_context=args.instructions
        )
    
    # Add podcast config if needed
    if args.format == "podcast":
        config_dict["podcast_config"] = PodcastConfig(
            num_speakers=args.speakers if hasattr(args, 'speakers') else 2
        )
    
    # Add fiction config if needed
    if args.format == "fiction":
        config_dict["fiction_config"] = FictionConfig(
            genre=args.genre if hasattr(args, 'genre') else "sci-fi"
        )
    
    return PipelineConfig(**config_dict)


async def main():
    """Main entry point"""
    
    parser = argparse.ArgumentParser(
        description="Transform GitHub repositories into audio content",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog="""
Examples:
  # Basic podcast from repo
  python repo_to_audio.py https://github.com/user/repo --format podcast

  # Long audiobook with custom config
  python repo_to_audio.py https://github.com/user/repo --format audiobook --length long

  # Fiction adaptation
  python repo_to_audio.py https://github.com/user/repo --format fiction --tone dramatic

  # Use ElevenLabs GenFM directly
  python repo_to_audio.py https://github.com/user/repo --path elevenlabs

  # With custom instructions
  python repo_to_audio.py https://github.com/user/repo --instructions "Focus on the security aspects"
        """
    )
    
    # Required arguments
    parser.add_argument("url", help="GitHub repository URL or content URL")
    
    # Output format
    parser.add_argument(
        "--format", "-f",
        choices=[f.value for f in OutputFormat],
        default="podcast",
        help="Output format (default: podcast)"
    )
    
    # Length
    parser.add_argument(
        "--length", "-l",
        choices=[l.value for l in ContentLength],
        default="medium",
        help="Content length (default: medium)"
    )
    
    # Tone
    parser.add_argument(
        "--tone", "-t",
        choices=[t.value for t in Tone],
        default="conversational",
        help="Content tone (default: conversational)"
    )
    
    # Audience
    parser.add_argument(
        "--audience", "-a",
        choices=[a.value for a in AudienceLevel],
        default="intermediate",
        help="Target audience level (default: intermediate)"
    )
    
    # Generation path
    parser.add_argument(
        "--path", "-p",
        choices=[p.value for p in ScriptGenerationPath],
        default="claude",
        help="Script generation path (default: claude)"
    )
    
    # Output file
    parser.add_argument(
        "--output", "-o",
        help="Output file path for audio"
    )
    
    # Config file
    parser.add_argument(
        "--config", "-c",
        help="Path to YAML configuration file"
    )
    
    # Custom instructions
    parser.add_argument(
        "--instructions", "-i",
        help="Custom instructions for content generation"
    )
    
    # Podcast-specific
    parser.add_argument(
        "--speakers",
        type=int,
        default=2,
        help="Number of speakers for podcast format"
    )
    
    # Fiction-specific
    parser.add_argument(
        "--genre",
        default="sci-fi",
        help="Genre for fiction format"
    )
    
    # Debug/verbose
    parser.add_argument(
        "--verbose", "-v",
        action="store_true",
        help="Enable verbose output"
    )
    
    args = parser.parse_args()
    
    if args.verbose:
        logging.getLogger().setLevel(logging.DEBUG)
    
    # Verify environment variables
    if not os.environ.get("ANTHROPIC_API_KEY"):
        console.print("[bold red]Error: ANTHROPIC_API_KEY environment variable not set[/bold red]")
        sys.exit(1)
    
    if not os.environ.get("ELEVENLABS_API_KEY"):
        console.print("[bold red]Error: ELEVENLABS_API_KEY environment variable not set[/bold red]")
        sys.exit(1)
    
    # Create configuration
    config = create_config_from_args(args)
    
    # Run pipeline
    pipeline = RepoToAudioPipeline(config)
    result = await pipeline.run()
    
    # Output result
    if result.get("success"):
        console.print("\n[bold green]🎉 Pipeline completed successfully![/bold green]")
        
        if "script" in result:
            script_preview = result["script"][:500] + "..." if len(result["script"]) > 500 else result["script"]
            console.print(Panel(script_preview, title="Script Preview"))
        
        if "project_id" in result.get("stages", {}).get("audio", {}):
            console.print(f"\n📁 ElevenLabs Project ID: {result['stages']['audio']['project_id']}")
    else:
        console.print("\n[bold red]❌ Pipeline failed[/bold red]")
        if "error" in result:
            console.print(f"Error: {result['error']}")
        sys.exit(1)


if __name__ == "__main__":
    asyncio.run(main())
